创建函数的4种方法

1. var func = function(){}


var func = function(){
  var a = 1;
  this.a = 2;
  return a +this.a;
}
console.log(typeof(func));//function
console.log(func.constructor);//function Function() { [native code] }
console.log(func.constructor());function anonymous(){}

每个对象都有constructor属性，func.constructor属性返回的是对象的构造函数，func.constructor()则是执行这个函数

2. var func = new function(){}


var func = new function(){
  var a = 1;
  this.a = 2;
  return a+this.a;
}
consloe.log(typeof(func));//object
console.log(func.constructor);
/*function (){

注意：执行var func = new function(){}时，进行了4步操作。 

1. var func = {};
2. func._proto_ = Function.prototype;//此时建立了远行链：func->Function.prototype
 ->Object.prototype->null
3. var result = Function.call(func);
4. return typeof result ==='func' ? result : obj;
